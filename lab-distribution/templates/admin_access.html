Prerequisites: install the AWS CLI v2 and kubectl
macOS 
# AWS CLI v2  (or download https://awscli.amazonaws.com/AWSCLIV2.pkg)
brew install awscli
brew install kubectl
aws --version && kubectl version --client
Windows (PowerShell) 
# AWS CLI v2  (or download https://awscli.amazonaws.com/AWSCLIV2.msi)
winget install -e --id Amazon.AWSCLI
winget install -e --id Kubernetes.kubectl
aws --version; kubectl version --client
Linux (x86_64) 
curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o awscliv2.zip
unzip awscliv2.zip && sudo ./aws/install
curl -LO "https://dl.k8s.io/release/$(curl -L -s https://dl.k8s.io/release/stable.txt)/bin/linux/amd64/kubectl"
sudo install -o root -g root -m 0755 kubectl /usr/local/bin/kubectl
aws --version && kubectl version --client
